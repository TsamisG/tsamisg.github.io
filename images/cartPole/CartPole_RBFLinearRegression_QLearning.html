def eps_greedy(a, env, eps=0.1):
    if np.random.random() > eps:
        return a
    else:
        return np.random.randint(env.action_space.n)

def gather_samples_s():
    samples = np.vstack((
        np.random.uniform(-2.4,2.4,size=N_SAMPLES),
        np.random.uniform(-2,2,size=N_SAMPLES),
        np.random.uniform(-0.2,0.2,size=N_SAMPLES),
        np.random.uniform(-1,1,size=N_SAMPLES)
    )).T
    
    return samples

ep_rewards = []
for ep in range(TRAIN_EPISODES):
    step=0
    done=False
    trun=False
    s = env.reset()[0]
    ep_reward = 0
    while (step<MAX_STEPS) and (not done) and (not trun):
        Q = []
        f = featurizer.transform(scaler.transform([s]))
        for action in range(env.action_space.n):
            Q.append(models[action].predict(f))

        a = eps_greedy(np.argmax(Q), env, eps=EPS)

        s2, r, done, trunc, _ = env.step(a)
        
        ep_reward += r
        step+=1
        if done == True:
            G = r
        else:
            Q = []
            f = featurizer.transform(scaler.transform([s2]))
            for action in range(env.action_space.n):
                Q.append(models[action].predict(f))
            G = r + GAMMA*np.max(Q)
        f = featurizer.transform(scaler.transform([s]))
        
        Ghat = models[a].predict(f)
        gradW = models[a].gradW(G, Ghat, f)
        gradb = models[a].gradb(G, Ghat)
        cacheW[models[a]] = DECAY * cacheW[models[a]] + (1 - DECAY) * gradW * gradW
        cacheb[models[a]] = DECAY * cacheb[models[a]] + (1 - DECAY) * gradb * gradb
        vW[models[a]] = MU * vW[models[a]] + (1 - MU) * ALPHA * gradW/(np.sqrt(cacheW[models[a]]) + 1e-08)
        vb[models[a]] = MU * vb[models[a]] + (1 - MU) * ALPHA * gradb/(np.sqrt(cacheb[models[a]]) + 1e-08)
        models[a].W -= vW[models[a]]
        models[a].b -= vb[models[a]]
        
        s = s2

    if (ep + 1) == 1 :
        print(f'Episode: [ {ep+1} / {TRAIN_EPISODES} ] | Reward: {ep_reward:.2f}')
    elif (ep + 1) % 20 == 0:
        print(f'Episode: [ {ep+1} / {TRAIN_EPISODES} ] | Mean Reward of the Last 20 Episodes: '
        f'{np.mean(ep_rewards[-20:]):.2f}')
    
    ep_rewards.append(ep_reward)
env.close()
